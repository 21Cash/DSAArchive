#include <stdio.h>
#include <math.h>


int main() {
    
    int n1, n2;
    printf("Enter n1 and n2: ");
    scanf("%d %d", &n1, &n2);
    
    
    int fPrime = -1;
    int lPrime = -1;
    
    int i, j;
    
    //Finding First Prime Number
    for(i = n1+1; i <= n2-1; i++) {
        int isPrime = 1;
        for(int j = 2; j < i; j++) {
            if(i % j == 0) {
                isPrime = 0;
            }
        } 
        if(isPrime) {
            fPrime = i;
            break;
        }       
    }
    
    
    // Finding Last Prime
    for(int i = n2-1; i >= n1; i--) {
        int isPrime = 1;
        for(int j = 2; j < i; j++) {
            if(i % j == 0) {
                isPrime = 0;
            }
        } 
        if(isPrime) {
            lPrime = i;
            break;
        } 
    }
    
    printf("Prime Number Is: %d\n", fPrime);
    
    
    // Finding Armstrong Number Between First Prime And Second Prime
    for(int i = fPrime; i <= lPrime; i++) {
        
        int num = i;
        
        int digits = 0;
        
        int temp = i;
        while(temp) {
            temp /= 10;
            digits++;
        }
        
        int sum = 0;
        temp = i;
        while(temp) {
            
            int t = temp % 10;
            temp /= 10;
            sum += pow(t, digits);
        }
        if(sum == num) {
            printf("Armstrong Number: %d\n", num);
            break;   
        }
    }
    
    return 0;
    
}
